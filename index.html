🎨 โปรเจกต์เว็บคำนวณเกรดเฉลี่ย

โปรแกรมคำนวณเกรดเฉลี่ยสะสม (GPAX) พัฒนาด้วย JavaScript ช่วยให้นักเรียนคำนวณคะแนนได้ง่ายขึ้น

// ฟังก์ชันคำนวณเกรดเฉลี่ย
function calculateGPA(scores) {
  let total = scores.reduce((a, b) => a + b, 0);
  return total / scores.length;
}
